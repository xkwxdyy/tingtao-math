\begin{spacing}{1.7}%%jianju
\begin{enumerate}



  \item [\textbf{1.1}] 设 $(X, \mathscr{T})$ 为拓扑空间, 其中
  \[\mathscr{T}=\left\{X, \emptyset, A,B\right\},\]
  这里 $A,B$ 为 $X$ 的不同非空真子集.  请问: $A$ 与 $B$ 由什么关系?

    \item [\textbf{1.2}] 设 $X,Y$为非空集合, $f: X\to Y$ 为映射, $\mathscr{T}^{\ast}$ 为 $Y$ 上的一个拓扑. 定义
    \[\mathscr{T}=\left\{f^{-1}\left(G\right): G\in \mathscr{T}^{\ast}\right\}.\]
    证明: $\mathscr{T}$ 是 $X$ 上的一个拓扑.


      \item [\textbf{1.3}] 令
      \[\mathscr{T}=\left\{(a,+\infty):a\in\mathbf{R}\right\}\cup\left\{\mathbf{R}, \emptyset\right\}.\]
      证明:  $\mathscr{T}$ 是 $\mathbf{R}$ 上的一个拓扑.

  \item [\textbf{1.4}] 设
$$\mathscr{T}=\{E_n=\{n, n+1, n+2, \cdots\}: n\in \mathbf{N}_+\}\bigcup\{\emptyset\}$$
\begin{itemize}
\item [(1)] 请证明$\mathscr{T}$是$\mathbf{N}_+$上的拓扑.
\item [(2)] 请写出$(\mathbf{N}_+,\mathscr{T})$中的所有闭集.
\end{itemize}

  \item [\textbf{1.5}] 设  $(X, \mathscr{T})$ 为拓扑空间, $a\notin X$. 令
  \[X^{\ast}=X\cup \left\{a\right\},\ \mathscr{T}^{\ast}=\mathscr{T}\cup \left\{X^{\ast}\right\}.\]
  证明: $\mathscr{T}^{\ast}$ 是 $X^{\ast}$ 上的一个拓扑.


  \item [\textbf{1.6}] 设  $(X, \mathscr{T})$ 为拓扑空间, $a\notin X$. 令
  \[X^{\ast}=X\cup \left\{a\right\},\ \mathscr{T}^{\ast}=\left\{\left\{a\right\}\cup U: U\in\mathscr{T}\right\}\cup \left\{\emptyset\right\}.\]
  请问: $\mathscr{T}^{\ast}$ 是 $X^{\ast}$ 上的的拓扑吗 ? 请说明理由.



   \item [\textbf{1.7}] 设 $X$ 为非空集合,
   \[\mathscr{T}=\left\{A: A\subset X,\  {\color{red}\sharp
(A^c)<\infty}\right\}\cup \left\{\emptyset\right\}.\]
证明: $\mathscr{T}$ 是 $X$ 上的一个拓扑. 该拓扑称为$X$上的{\color{red}余有限拓扑}.


     \item [\textbf{1.8}] 设 $X$ 为非空集合,
   \[\mathscr{T}=\left\{A: A\subset X,\  {\color{red}\sharp
(A^c)\leqslant \aleph_0}\right\}\cup \left\{\emptyset\right\}.\]
证明: $\mathscr{T}$ 是 $X$ 上的一个拓扑. 该拓扑称为$X$上的{\color{red}余可数拓扑}.


      \item [\textbf{1.9}] 设  $X$ 为非空集合, $\mathscr{T}_{\gamma}(\gamma\in\Gamma)$ 为$X$上的一族拓扑.  证明: $\displaystyle \bigcap_{\gamma\in\Gamma}\mathscr{T}_{\gamma}$ 也是 $X$ 上的一个拓扑.

       \item [\textbf{1.10}] 设  $X$ 为非空集合, $\mathscr{T}_1, \mathscr{T}_2$ 为$X$上的两个不同拓扑. 请问: $\mathscr{T}_1\cup\mathscr{T}_2$ 一定也是 $X$ 上的一个拓扑吗？ 请说明理由.


    \item [\textbf{1.11}] 设$X=\{a,b,c\}$. 请写出$X$上的所有拓扑.

\end{enumerate}
